#include <stdio.h>
#define max 5
int queue[max];
int front = -1;
int rear = -1;
void enqueue(){
    int item;
    printf("enter a num:");
    scanf("%d",&item);
    if(rear == max-1){
        printf("Overflow!");
    }
    else if(front == -1 && rear == -1){
        front = rear = 0;
        queue[rear] = item;
        printf("%d is inserted\n",item);
    }
    else{
        queue[++rear]= item;
        printf("%d is inserted\n",item);
    }
}
void dequeue(){
    int value;
    if(front == -1){
        printf("underflow");
    }
    else{
        
        value = queue[front];
        if(front == rear){
            front = rear = -1;
        }
        else{
          front++;  
        }
        printf("%d is deleted\n",value);
    }
}
void display(){
    if(front == -1){
        printf("Stck undderflow");
    }
    else{
        printf("Elements!!!\n");
        for(int i=front;i<=rear;i++){
            printf("%d->%d\n",i,queue[i]);
        }
    }
}
int main(){
    enqueue();
    enqueue();
    enqueue();
    enqueue();
    display();
    dequeue();
    display();
}
